### Create tbl UOM

create table UOM(
    Id INT(11) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    Code VARCHAR(120),
    Name VARCHAR(120),
    Remark VARCHAR(255),
    Status TINYINT(1) DEFAULT 1,
    CreateAt timestamp NOT NULL DEFAULT current_timestamp()
    );
